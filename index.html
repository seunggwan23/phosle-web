import streamlit as st import numpy as np import plotly.graph_objects as go st.set_page_config(page_title="건물 에너지 AI", page_icon="⚡", layout="wide") st.markdown(""" """, unsafe_allow_html=True) # ────────────────────────────────────────────────────────────────── # 에너지 계산 함수 (보정 계수 기반 현실적 모델) # 기준: 한국 업무용 건물 평균 에너지 강도 150 kWh/m²/년 # ────────────────────────────────────────────────────────────────── BASE_INTENSITY_KWH_M2 = 150.0 # kWh/m²/년 (국토교통부 에너지 기준) ELEC_RATE_WON = 130 # 원/kWh (한전 일반용 평균 단가) PYEONG_TO_M2 = 3.3058 # 1평 = 3.3058 m² # 월별 에너지 분포 비중 (한국 기후 기준: 여름/겨울 집중) MONTHLY_WEIGHTS = np.array([0.10, 0.09, 0.07, 0.06, 0.07, 0.10, 0.12, 0.12, 0.08, 0.07, 0.06, 0.06]) MONTHLY_WEIGHTS = MONTHLY_WEIGHTS / MONTHLY_WEIGHTS.sum() # 합계 1로 정규화 def calc_annual_kwh(pyeong: float, wwr: float, insulation_mm: float, hvac_setpoint: float) -> float: """ 보정 계수 기반 연간 에너지 소비량(kWh) 계산. - 평수(pyeong)에 선형 비례 → 두 배 크면 두 배 소비 - WWR: 0.3 기준, ±1% 변화당 냉방 부하 2% 변동 (현실적 범위: 최대 ±40%) - 단열재: 150mm 기준, 50mm 감소당 난방 부하 8% 증가 (현실적 범위: 50mm→+24%, 400mm→-20%) - HVAC 온도: 24°C 기준, 1°C 낮출수록 냉방 4% 증가 (현실적 범위: 18°C→+24%, 30°C→-24%) - 각 보정값은 ±35% 이내로 클램핑 → 극단값 방지 """ floor_m2 = pyeong * PYEONG_TO_M2 # 창면적비 보정 (0.3 기준) wwr_delta = np.clip((wwr - 0.3) * 2.0, -0.35, 0.35) # 단열재 보정 (150mm 기준, 얇을수록 에너지 증가) ins_delta = np.clip((150 - insulation_mm) / 150 * 0.30, -0.30, 0.30) # HVAC 설정온도 보정 (24°C 기준) hvac_delta = np.clip((24 - hvac_setpoint) * 0.04, -0.25, 0.25) # 최종 에너지 강도 (kWh/m²/년) intensity = BASE_INTENSITY_KWH_M2 * (1 + wwr_delta + ins_delta + hvac_delta) return intensity * floor_m2 def monthly_breakdown(annual_kwh: float) -> np.ndarray: """월별 에너지(kWh) 배열 반환""" return MONTHLY_WEIGHTS * annual_kwh def won_to_manwon(won: float) -> str: """원 → 만원 단위 문자열 (예: 234만원)""" manwon = won / 10000 if manwon >= 1000: return f"{manwon/100:.0f}억원" return f"{manwon:,.0f}만원" # ────────────────────────────────────────────────────────────────── # 사이드바 # ────────────────────────────────────────────────────────────────── with st.sidebar: st.markdown("### ⚙️ 건물 조건 설정") st.caption("슬라이더를 조절하면 결과가 즉시 바뀝니다") st.markdown("---") st.markdown("**🏢 건물 규모**") pyeong = st.slider("건물 면적 (평)", 10, 1000, 100, 10, help="1평 ≈ 3.3m². 아파트 30평, 소형 사무실 50평, 중형 빌딩 300평 기준") st.markdown("**🔧 건물 사양**") wwr = st.slider("🪟 창면적비 (WWR)", 0.1, 0.8, 0.3, 0.05, help="창문이 외벽에서 차지하는 비율. 높을수록 여름 냉방 부하 증가") insulation_mm = st.slider("🧱 단열재 두께 (mm)", 50, 400, 150, 10, help="두꺼울수록 냉난방 에너지 손실 감소") hvac_setpoint = st.slider("🌡️ 냉난방 설정온도 (°C)", 18, 30, 24, 1, help="여름 냉방 기준. 낮을수록 냉방 부하 증가") st.markdown("---") floor_m2 = pyeong * PYEONG_TO_M2 st.markdown(f"""
📐 면적: {pyeong}평 ({floor_m2:.0f} m²)
📌 기준: WWR 0.30 / 단열 150mm / 24°C
""", unsafe_allow_html=True) # ────────────────────────────────────────────────────────────────── # 핵심 계산 # ────────────────────────────────────────────────────────────────── # 현재 설정 & 기준 설정 에너지 kwh_now = calc_annual_kwh(pyeong, wwr, insulation_mm, hvac_setpoint) kwh_base = calc_annual_kwh(pyeong, 0.3, 150, 24) # 기준 조건 diff_kwh = kwh_base - kwh_now # + 이면 절감, - 이면 증가 diff_pct = diff_kwh / kwh_base * 100 # 비용 cost_now = kwh_now * ELEC_RATE_WON # 원/년 cost_base = kwh_base * ELEC_RATE_WON cost_diff = abs(cost_base - cost_now) # 원/년 절감 또는 증가 is_saving = diff_kwh >= 0 verb = "절감" if is_saving else "증가" color_main = "#50C878" if is_saving else "#F87171" color_bg = "result-good" if is_saving else "result-bad" # 효율 등급 if diff_pct >= 20: grade, gc = "A+", "grade-A" elif diff_pct >= 10: grade, gc = "A", "grade-A" elif diff_pct >= 0: grade, gc = "B", "grade-B" elif diff_pct >= -10: grade, gc = "C", "grade-C" else: grade, gc = "D", "grade-D" # 월별 데이터 monthly_now = monthly_breakdown(kwh_now) / 1000 # MWh monthly_base = monthly_breakdown(kwh_base) / 1000 # MWh # 계절별 집계 summer_idx = [5, 6, 7] # 6,7,8월 (0-indexed) winter_idx = [11, 0, 1] # 12,1,2월 kwh_summer_now = float(monthly_now[summer_idx].sum()) kwh_winter_now = float(monthly_now[winter_idx].sum()) kwh_summer_base = float(monthly_base[summer_idx].sum()) kwh_winter_base = float(monthly_base[winter_idx].sum()) # ────────────────────────────────────────────────────────────────── # 헤더 # ────────────────────────────────────────────────────────────────── st.markdown(f"""
⚡ 건물 에너지 절감 시뮬레이터
현재 설정: {pyeong}평  ·  창면적비 {wwr:.0%}  ·  단열 {insulation_mm}mm  ·  설정온도 {hvac_setpoint}°C
""", unsafe_allow_html=True) # ────────────────────────────────────────────────────────────────── # 핵심 결과 배너 (가장 중요한 정보) # ────────────────────────────────────────────────────────────────── annual_str = won_to_manwon(cost_diff) monthly_str = won_to_manwon(cost_diff / 12) st.markdown(f"""
기준 조건 대비  ·  연간 에너지 {verb}
{abs(diff_kwh)/1000:,.1f} MWh/년
💰 연간  {annual_str}  {verb}
월 평균 {monthly_str} {verb}
현재 연간 전기요금 추정: {won_to_manwon(cost_now)}/년  (기준: {won_to_manwon(cost_base)}/년)
{grade}
에너지 효율 등급
""", unsafe_allow_html=True) # ────────────────────────────────────────────────────────────────── # KPI 3개 (여름·겨울·연간) # ────────────────────────────────────────────────────────────────── sc = "#50C878" if kwh_summer_base >= kwh_summer_now else "#F87171" wc = "#50C878" if kwh_winter_base >= kwh_winter_now else "#F87171" sv = "▼" if kwh_summer_base >= kwh_summer_now else "▲" wv = "▼" if kwh_winter_base >= kwh_winter_now else "▲" st.markdown(f"""
☀️ 여름철 냉방 에너지
{kwh_summer_now:.1f} MWh
{sv} 기준 대비 {abs(kwh_summer_base-kwh_summer_now):.1f} MWh
❄️ 겨울철 난방 에너지
{kwh_winter_now:.1f} MWh
{wv} 기준 대비 {abs(kwh_winter_base-kwh_winter_now):.1f} MWh
📅 연간 총 에너지
{kwh_now/1000:.1f} MWh
기준 {kwh_base/1000:.1f} MWh  ·  {floor_m2:.0f} m²
""", unsafe_allow_html=True) # ────────────────────────────────────────────────────────────────── # 차트(좌) + 인사이트(우) # ────────────────────────────────────────────────────────────────── col1, col2 = st.columns([3, 2]) with col1: MONTHS = ["1월","2월","3월","4월","5월","6월","7월","8월","9월","10월","11월","12월"] DARK, CARD, GRID = "#0A0F1E", "#0D1526", "#1A2740" fig = go.Figure() fig.add_trace(go.Bar( x=MONTHS, y=monthly_base, name="기준 조건 (WWR 0.3 / 단열 150mm / 24°C)", marker_color="rgba(107,138,176,0.35)", hovertemplate="%{x} 기준: %{y:.2f} MWh" )) fig.add_trace(go.Bar( x=MONTHS, y=monthly_now, name="현재 설정", marker=dict(color=[color_main]*12), hovertemplate="%{x} 현재: %{y:.2f} MWh" )) fig.update_layout( paper_bgcolor=DARK, plot_bgcolor=CARD, font=dict(family="Inter", color="#E2E8F0"), title=dict(text="📅 월별 에너지 소비량 비교", font=dict(size=14)), xaxis=dict(gridcolor=GRID, linecolor=GRID), yaxis=dict(gridcolor=GRID, linecolor=GRID, title="에너지 (MWh)"), barmode="group", bargap=0.25, legend=dict(bgcolor="rgba(0,0,0,0)", font=dict(color="#94A3B8"), orientation="h", yanchor="bottom", y=1.02, xanchor="left", x=0), margin=dict(l=16, r=16, t=64, b=16), height=360, ) st.plotly_chart(fig, use_container_width=True) with col2: st.markdown("#### 💡 개선 포인트") tips = [] # 창면적비 if wwr > 0.5: saving_kwh = calc_annual_kwh(pyeong, 0.4, insulation_mm, hvac_setpoint) - kwh_now extra_cost = abs(saving_kwh) * ELEC_RATE_WON tips.append(("warn", f"창면적비 {wwr:.0%}가 높습니다.
" f"0.4로 낮추면 연간 {won_to_manwon(extra_cost)} 절감 가능.
" f"→ 외부 차양(루버/필름) 설치를 권장합니다.")) elif wwr < 0.2: tips.append(("ok", f"창면적비 {wwr:.0%} — 냉방 부하 최소화.
" f"단, 자연채광 부족으로 조명 에너지가 늘 수 있습니다.")) else: tips.append(("ok", f"창면적비 {wwr:.0%}는 적정 범위입니다.")) # 단열재 if insulation_mm < 100: saving_kwh = kwh_now - calc_annual_kwh(pyeong, wwr, 150, hvac_setpoint) extra_cost = abs(saving_kwh) * ELEC_RATE_WON tips.append(("warn", f"단열재 {insulation_mm}mm는 너무 얇습니다.
" f"150mm로 강화하면 연간 {won_to_manwon(extra_cost)} 절감 가능.
" f"→ 외단열 시스템 적용을 검토하세요.")) elif insulation_mm >= 250: tips.append(("ok", f"단열재 {insulation_mm}mm — 고성능 단열 확보.
" f"냉난방 에너지 손실이 최소화됩니다.")) else: tips.append(("ok", f"단열재 {insulation_mm}mm는 적정 수준입니다.")) # HVAC if hvac_setpoint < 22: saving_kwh = kwh_now - calc_annual_kwh(pyeong, wwr, insulation_mm, 24) extra_cost = abs(saving_kwh) * ELEC_RATE_WON tips.append(("warn", f"설정온도 {hvac_setpoint}°C는 낮습니다.
" f"24°C로 올리면 연간 {won_to_manwon(extra_cost)} 절감 가능.
" f"→ 여름 냉방 권장 온도는 26°C입니다.")) elif hvac_setpoint > 26: saving_kwh = kwh_now - calc_annual_kwh(pyeong, wwr, insulation_mm, 24) extra_cost = abs(saving_kwh) * ELEC_RATE_WON tips.append(("warn", f"설정온도 {hvac_setpoint}°C는 높습니다.
" f"24°C로 낮추면 연간 {won_to_manwon(extra_cost)} 절감 가능.")) else: tips.append(("ok", f"설정온도 {hvac_setpoint}°C는 적정 범위입니다.")) for cls, txt in tips: st.markdown(f'
